npm install  :
docker run -it --rm \
  -v "$PWD:/app" \
  -w /app \
  -p 3000:3000 \
  node:alpine \
  npm install

npm run dev :
docker run -it --rm \
  -v "$PWD:/app" \
  -w /app \
  -p 3000:3000 \
  node:alpine \
  npm run dev

shell bash :
docker run -it --rm \
  -v "$PWD:/app" \
  -w /app \
  -p 3000:3000 \
  node:latest \
  bash
